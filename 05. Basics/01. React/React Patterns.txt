ğŸ”¹ 10. Custom Hooks

	Reuse logic across components

function useWindowWidth() {
  const [width, setWidth] = useState(window.innerWidth);
  useEffect(() => {
    window.onresize = () => setWidth(window.innerWidth);
  }, []);
  return width;
}
